    <script>

        // Countdown Timer

        function updateCountdown() {

            const weddingDate = new Date('2025-02-02T14:00:00');

            const now = new Date();

            const difference = weddingDate - now;


            const days = Math.floor(difference / (1000 * 60 * 60 * 24));

            const hours = Math.floor((difference / (1000 * 60 * 60)) % 24);

            const minutes = Math.floor((difference / 1000 / 60) % 60);

            const seconds = Math.floor((difference / 1000) % 60);


            document.getElementById('days').innerHTML = `<div class="text-4xl font-bold">${days}</div><div>天</div>`;

            document.getElementById('hours').innerHTML = `<div class="text-4xl font-bold">${hours}</div><div>時</div>`;

            document.getElementById('minutes').innerHTML = `<div class="text-4xl font-bold">${minutes}</div><div>分</div>`;

            document.getElementById('seconds').innerHTML = `<div class="text-4xl font-bold">${seconds}</div><div>秒</div>`;

        }


        setInterval(updateCountdown, 1000);

        updateCountdown();


        // Photo Gallery

        let currentSlide = 0;

        const photos = Array(10).fill('/api/placeholder/800/600');

        const gallery = document.getElementById('gallery');


        function updateGallery() {

            gallery.innerHTML = `

                <img src="${photos[currentSlide]}" 

                     alt="Wedding photo ${currentSlide + 1}" 

                     class="w-full h-full object-cover rounded-lg">

            `;

        }


        function nextSlide() {

            currentSlide = (currentSlide + 1) % photos.length;

            updateGallery();

        }


        function prevSlide() {

            currentSlide = (currentSlide - 1 + photos.length) % photos.length;

            updateGallery();

        }


        updateGallery();
